int len1 = num1.size(), len2 = num2.size();
    if(len1 > len2) return 1;
    else if(len2 > len1) return 2;
    else{
        int i = 0, j = 0;
        while(i < len1 && j < len2 && num1[i] == num2[j]){
            i++;
            j++;
        }
        if(i >= len1) return 0;
        else if(num1[i]-'0' > num2[j]-'0') return 1;
        else if(num1[i]-'0' < num2[j]-'0') return 2;
    }
    return -1;
